import tkinter as tk from nltk.tokenize import word_tokenize from textblob import TextBlob import newspaper def summarize(): url = urlInput.get() article = newspaper.Article(url) article.download() article.parse() titleOutput.value = article.title authorOutput.value = article.authors[0] publicationOutput.value = article.publish_date # Generate a summary of the article summary_text = TextBlob(article.text).sentences[0:5] summary_text = ' '.join([str(sentence) for sentence in summary_text]) summaryOutput.value = summary_text # Create and configure the Tkinter elements root = tk.Tk() root.title('Summarizer') urlInput = tk.Entry(root) summarizeButton = tk.Button(root, text='Summarize', command=summarize) titleOutput = tk.Label(root) authorOutput = tk.Label(root) publicationOutput = tk.Label(root) summaryOutput = tk.Text(root, height=10, width=60) urlInput.pack() summarizeButton.pack() titleOutput.pack() authorOutput.pack() publicationOutput.pack() summaryOutput.pack() # Start the Tkinter mainloop root.mainloop()
